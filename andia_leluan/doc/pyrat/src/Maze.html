414    def locations_to_action ( self:   Self,
415                              source: Integral,
416                              target: Integral
417                            ) ->      Optional[Action]: 
418
419        """
420            Function to transform two locations into an action to reach the target from the source.
421            In:
422                * self:   Reference to the current object.
423                * source: Vertex on which the player is.
424                * target: Vertex where the character wants to go.
425            Out:
426                * action: Action to go from the source to the target, or None if the move is impossible.
427        """
428
429        # Debug
430        assert isinstance(source, Integral), "Argument 'source' must be an integer"
431        assert isinstance(target, Integral), "Argument 'target' must be an integer"
432        assert self.i_exists(source), "Source is not in the maze"
433        assert self.i_exists(target), "Target is not in the maze"
434
435        #Get the coordinates difference
436        difference = self.coords_difference(source, target)
437
438        # Translate in a move
439        if difference == (0, 0):
440            action = Action.NOTHING
441        elif difference == (0, -1):
442            action = Action.WEST
443        elif difference == (0, 1):
444            action = Action.EAST
445        elif difference == (1, 0):
446            action = Action.SOUTH
447        elif difference == (-1, 0):
448            action = Action.NORTH
449        else:
450            action = None
451        return action
